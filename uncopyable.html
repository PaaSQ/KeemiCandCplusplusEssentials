#include <string>

class Uncopyable {

protected:                                 
	Uncopyable() {} 
	~Uncopyable() {}

private:
	Uncopyable(const Uncopyable&);
	Uncopyable& operator=(const Uncopyable&);

};

//
class Sub : public Uncopyable {
public:
	Sub( const std::string& str)
		:str_(str) {
	}
	// ...
private:
	std::string str_;
};

//
int main(int argc, char *argv[])
{
	Sub instance1(std::string("Hello"));
	Sub instance2(std::string("Hello"));
	instance1 = instance2;
}
