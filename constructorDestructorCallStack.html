#include <iostream>

class TimeKeeper {

public:
	TimeKeeper() {
		std::cout << "TimeKeeper()" << std::endl;
	}

	virtual ~TimeKeeper() {
		std::cout << "~TimeKeeper()"  << std::endl;;
	}
};

class AtomicClock: public TimeKeeper {

public:
	AtomicClock() {
		std::cout << "AtomicClock()"  << std::endl;;
	}

	virtual ~AtomicClock() {
		std::cout << "~AtomicClock()"  << std::endl;;
	}
};

class AtomicClockSub: public AtomicClock { 

public:
	AtomicClockSub() {
		std::cout << "AtomicClockSub()"  << std::endl;;
	}

	virtual ~AtomicClockSub() {
		std::cout << "~AtomicClockSub()"  << std::endl;;
	}
};

//
int main(int argc, char *argv[])
{
	TimeKeeper *timeKeeper = new AtomicClockSub();
	// ...
	delete timeKeeper;
}
